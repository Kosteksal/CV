Code example

                            
                                import i18Obj from './translate.js';
        
                                const en = document.querySelector('.en');
                                const ru = document.querySelector('.ru');
                                const langSw = document.querySelectorAll('.lang');
                                const trans = document.querySelectorAll('[data-i18]');
                                
                                langSw.forEach(i => i.addEventListener('click', getTranslateSw));
                                
                                function getTranslateSw (event) {
                                    if(event.target.classList.contains('ru')) {
                                        langSw.forEach((i) => i.classList.remove('active'));
                                        event.target.classList.add('active');  
                                        getTranslate(ru);
                                    }
                                    if(event.target.classList.contains('en')) {
                                        langSw.forEach((i) => i.classList.remove('active'));
                                        event.target.classList.add('active'); 
                                        getTranslate(en);
                                    }
                                };
                                
                                function getTranslate(lang) {
                                    if(lang === ru) {
                                        trans.forEach(i => i.textContent = i18Obj['ru'][i.getAttribute('data-i18')] );
                                        langStore = 'ru';
                                        telPlace.placeholder = 'Телефон';
                                        textPlace.placeholder = 'Сообщение';
                                    }
                                    if(lang === en) {
                                        trans.forEach(i => i.textContent = i18Obj['en'][i.getAttribute('data-i18')] );
                                        langStore = 'en'; 
                                        telPlace.placeholder = 'Phone';
                                        textPlace.placeholder = 'Message';
                                    }
                                };
                                
                                const telPlace = document.querySelector('.cont-tel');
                                const textPlace = document.querySelector('.cont-text');
